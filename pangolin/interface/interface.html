104def rv_factory(op, *args) -> OperatorRV:
105    """
106    Given an op and some args, create an OperatorRV. By default, this just calls `OperatorRV`.
107    But this can be overridden if you'd like to mess with subclasses by putting a new function
108    on the `rv_factories` list.
109    """
110    return rv_factories[-1](op, *args)
111
112
113rv_factories = [OperatorRV]
114
115class SetRVFactory:
116    def __init__(self, rv_factory):
117        self.rv_factory = rv_factory
118
119    def __enter__(self):
120        rv_factories.append(self.rv_factory)
121
122    def __exit__(self, exc_type, exc_value, exc_tb):
123        assert rv_factories.pop() == self.rv_factory

137@api
138def makerv(x) -> OperatorRV:
139    """
140    In principle this is just a function that does nothing if the input is OperatorRV and creates a
141    new OperatorRV(Constant(...)) if it's something that can be cast to a numpy array. This
142    function is called by all the standard functions (`add`, `normal`, etc.) so that users can
143    directly pass constants.
144
145    This behavior, however, should be programmable, in case someone would like to use some other
146    class of `RV` (which must be a subclass of `OperatorRV`). So we create a list `makerv_funs` that
147    outside code can (presumably temporarily) stuck a new function on top of.
148    """
149    assert not non_operator_rv(x)
150    my_makerv = makerv_funs[-1]
151    out = my_makerv(x)
152    if not isinstance(out, OperatorRV):
153        raise Exception(
154            f"Custom makerv function must return subclass of OperatorRV was " f"{type(out)}"
155        )
156    return out
157
158def standard_makerv(x) -> OperatorRV:
159    """
160    The "standard" makerv function. If the input is `OperatorRV`, then it just returns it.
161    Otherwise, creates a new `op=Constant(x)` and then creates an `OperatorRV` using
162    `current_rv_factory()`
163    """
164    assert not non_operator_rv(x)
165
166    if isinstance(x, OperatorRV):
167        return x
168    else:
169        return rv_factory(Constant(x))
170
171
172makerv_funs = [standard_makerv]

323def implicit_vectorized_scalar_fun(fun):
324    def new_fun(*args):
325        args = tuple(makerv(p) for p in args)
326        in_axes = []
327        vec_shape = None
328        for p in args:
329            if p.shape == ():
330                # scalars are always OK
331                in_axes.append(None)
332            else:
333                if vec_shape:
334                    assert (
335                            p.shape == vec_shape
336                    ), "can only vectorize scalars + arrays of same shape"
337                else:
338                    vec_shape = p.shape
339                in_axes.append(0)
340        if vec_shape is None:
341            return fun(*args)
342        else:
343            from .vmap import vmap
344            vectorized_dims = len(vec_shape)
345            my_fun = fun
346            for i in range(vectorized_dims):
347                my_fun = vmap(my_fun, in_axes)
348            return my_fun(*args)
349    return new_fun

Matrix product of two arrays. The behavior follows that of numpy.matmul except that (currently) a and b must both be 1-D or 2-D arrays. In particular:

  • If a and b are both 1-D then this represents an inner-product.
