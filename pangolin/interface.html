139def makerv(x) -> OperatorRV:
140    """
141    In principle this is just a function that does nothing if the input is OperatorRV and creates a
142    new OperatorRV(Constant(...)) if it's something that can be cast to a numpy array. This
143    function is called by all the standard functions (`add`, `normal`, etc.) so that users can
144    directly pass constants.
145
146    This behavior, however, should be programmable, in case someone would like to use some other
147    class of `RV` (which must be a subclass of `OperatorRV`). So we create a list `makerv_funs` that
148    outside code can (presumably temporarily) stuck a new function on top of.
149    """
150    assert not non_operator_rv(x)
151    my_makerv = makerv_funs[-1]
152    out = my_makerv(x)
153    if not isinstance(out, OperatorRV):
154        raise Exception(
155            f"Custom makerv function must return subclass of OperatorRV was " f"{type(out)}"
156        )
157    return out
